/**
 * curl类函数
 */
//post方式提交获取数据
function curl_post($url='', $postdata='', $options=array()){
    $ch = curl_init($url);
    curl_setopt($ch, CURLOPT_RETURNTRANSFER, 1);
    curl_setopt($ch, CURLOPT_POST, 1);
 curl_setopt($ch, CURLOPT_SSL_VERIFYPEER, false);
    curl_setopt($ch, CURLOPT_SSL_VERIFYHOST, false);
    curl_setopt($ch, CURLOPT_POSTFIELDS, $postdata);
    curl_setopt($ch, CURLOPT_TIMEOUT, 5);
    if (!empty($options)){
        curl_setopt_array($ch, $options);
    }
    $data = curl_exec($ch);
    curl_close($ch);
    return $data;
}
//get方式提交获取数据
function curl_get($url='', $options=array()){
    $ch = curl_init($url);
    curl_setopt($ch, CURLOPT_RETURNTRANSFER, 1);
    curl_setopt($ch, CURLOPT_TIMEOUT, 5);
    if (!empty($options)){
        curl_setopt_array($ch, $options);
    }
    $data = curl_exec($ch);
    curl_close($ch);
    return $data;
}
////////获取信任码
function gettoken($app_id,$access_token,$timestamp,$app_secret){
$url = "http://api.189.cn/v2/dm/randcode/token?";

        $param['app_id']= "app_id=".$app_id;
        $param['access_token'] = "access_token=".$access_token;
        $param['timestamp'] = "timestamp=".$timestamp;
        ksort($param);
        $plaintext = implode("&",$param);
        $param['sign'] = "sign=".rawurlencode(base64_encode(hash_hmac("sha1", $plaintext, $app_secret, $raw_output=True)));
        ksort($param);
        $url .= implode("&",$param);
        $result = curl_get($url);
        $resultArray = json_decode($result,true);
        $token = $resultArray['token'];
  return $token;
}
////////发送短信方法
function send($token,$phone,$exp_time,$app_id,$access_token,$timestamp,$app_secret,$randcode){
  $url = "http://api.189.cn/v2/dm/randcode/sendSms";

        $param['app_id']= "app_id=".$app_id;
        $param['access_token'] = "access_token=".$access_token;
        $param['timestamp'] = "timestamp=".$timestamp;
        $param['token'] = "token=".$token;
        $param['phone'] = "phone=".$phone;
        $param['randcode'] = "randcode=".$randcode;
        if(isset($exp_time))
            $param['exp_time'] = "exp_time=".$exp_time;
        ksort($param);
        $plaintext = implode("&",$param);
        $param['sign'] = "sign=".rawurlencode(base64_encode(hash_hmac("sha1", $plaintext, $app_secret, $raw_output=True)));
        ksort($param);
        $str = implode("&",$param);
        $result = curl_post($url,$str);
        $resultArray = json_decode($result,true);
  return $resultArray;
}
/////////获取令牌
function getaccess_token($appid,$appsecret,$tokenAPI){
$post_data = array();
$post_data['app_id'] =$appid;
$post_data['app_secret'] =$appsecret;
$post_data['grant_type'] ="client_credentials";
$post_data['btn_ccauth'] ="认证授权";
$o="";
foreach ($post_data as $k=>$v)
{
    $o.= "$k=".urlencode($v)."&";
}
$post_data=substr($o,0,-1);
$result=json_decode(curl_post($tokenAPI,$post_data,null));
$access_token= $result->access_token;
return $access_token;
}

$appid = "1651132165464654";  ///你申请的应用ID

$tokenAPI = "https://oauth.api.189.cn/emp/oauth2/v2/access_token";
$phone="1131654648";  ///接收短信验证码号码
$randcode="456879";     ///随机生成验证码

$access_token= ($appid,$appsecret,$tokenAPI);
if(!$access_token){
echo "验证码发送失败！";exit;
}

$timestamp=date('Y-m-d H:i:s');
$token=gettoken($appid,$access_token,$timestamp,$appsecret);
if(!$token){
echo "验证码发送失败！";exit;
}

$result=send($token,$phone,$exp_time,$appid,$access_token,$timestamp,$appsecret,$randcode);
if(!$result['res_code']){
echo "验证码发送成功！";
}else{
echo "验证码发送失败！";
}
